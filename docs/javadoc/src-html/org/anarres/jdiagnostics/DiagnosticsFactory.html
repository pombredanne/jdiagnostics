001package org.anarres.jdiagnostics;
002
003/**
004 *
005 * @author shevek
006 */
007public class DiagnosticsFactory {
008
009    public static Query forException(ClassNotFoundException e) {
010        CompositeQuery query = new CompositeQuery();
011        query.add(new ClassExistsQuery(e.getMessage()));
012        query.add(new ClassLoaderQuery("system", String.class.getClassLoader()));
013        query.add(new ClassLoaderQuery("threadcontext", Thread.currentThread().getContextClassLoader()));
014        query.add(new ClassLoaderQuery("jdiagnostics", DiagnosticsFactory.class.getClassLoader()));
015        return query;
016    }
017}
