028/**
029 * Product Metadata and version information.
030 *
031 * This class is driven by metadata inserted into JAR files by the Netflix
032 * gradle-info-plugin, Maven Plexus or Bnd plugins, or similar.
033 *
034 * Any module built with appropriate metadata in the MANIFEST.MF will show up
035 * regardless of whether it is officially a component of the package software
036 * or merely a third party dependency.
037 *
038 * @author shevek
039 */

111    public static class ModuleVersion {
112
113        private final String group;
114        private final String name;
115        private final String version;
116
117        public ModuleVersion(@CheckForNull String group, @Nonnull String name, @Nonnull String version) {
118            this.group = group;
119            this.name = name;
120            this.version = version;
121        }
122
123        /**
124         * Returns what is probably the Maven group name of the artifact.
125         *
126         * @return what is probably the Maven group name of the artifact.
127         */
128        @CheckForNull
129        public String getGroup() {
130            return group;
131        }
132
133        /**
134         * Returns the name of the artifact, which might also be the maven artifact name.
135         *
136         * @return the name of the artifact, which might also be the maven artifact name.
137         */
138        @Nonnull
139        public String getName() {
140            return name;
141        }
142
143        @Nonnull
144        public String getVersion() {
145            return version;
146        }
147
148        private void toName(@Nonnull StringBuilder buf) {
149            if (group != null)
150                buf.append(group).append(':');
151            buf.append(name);
152        }
153
154        public String toName() {
155            StringBuilder buf = new StringBuilder();
156            toName(buf);
157            return buf.toString();
158        }
159
160        @Override
161        public String toString() {
162            StringBuilder buf = new StringBuilder();
163            toName(buf);
164            buf.append(':').append(version);
165            return buf.toString();
166        }
167
168        /** Parses Implementation-Title from gradle-info-plugin. */
169        @CheckForNull
170        public static ModuleVersion forGradleInfo(@Nonnull Attributes attributes) {
171            String implementationTitle = attributes.getValue(MetadataProperty.Implementation_Title.key);
172            if (implementationTitle == null)
173                return null;
174            int hashIdx = implementationTitle.indexOf('#');
175            if (hashIdx == -1)
176                return null;    // Not from Netflix plugin.
177            String group = implementationTitle.substring(0, hashIdx);
178            implementationTitle = implementationTitle.substring(hashIdx + 1);
179            int semiIdx = implementationTitle.indexOf(';');
180            if (semiIdx == -1)
181                return null;
182            String version = implementationTitle.substring(semiIdx + 1);
183            implementationTitle = implementationTitle.substring(0, semiIdx);
184            return new ModuleVersion(group, implementationTitle, version);
185        }
186
187        @CheckForNull
188        public static ModuleVersion forMavenPluexus(@Nonnull Attributes attributes) {
189            String implementationTitle = attributes.getValue(MetadataProperty.Implementation_Title.key);
190            if (implementationTitle == null)
191                return null;
192            String implementationVersion = attributes.getValue(MetadataProperty.Implementation_Version.key);
193            if (implementationVersion == null)
194                return null;
195            String implementationVendor = attributes.getValue(MetadataProperty.Implementation_Vendor_Id.key);
196            return new ModuleVersion(implementationVendor, implementationTitle, implementationVersion);
197        }
198
199        /** Parses Bundle-SymbolicName and Bundle-Version from Maven Bundle Plugin. */
200        @CheckForNull
201        public static ModuleVersion forMavenBundle(@Nonnull Attributes attributes) {
202            String bundleSymbolicName = attributes.getValue(MetadataProperty.Bundle_SymbolicName.key);
203            if (bundleSymbolicName == null)
204                return null;
205            String bundleVersion = attributes.getValue(MetadataProperty.Bundle_Version.key);
206            if (bundleVersion == null)
207                return null;
208            return new ModuleVersion(null, bundleSymbolicName, bundleVersion);
209        }
210
211        @CheckForNull
212        public static ModuleVersion forAttributes(@Nonnull Attributes attributes) {
213            // We can't guess the private resource name except by parsing the manifest.
214            // Luckily, Netflix puts the metadata into both places.
215            ModuleVersion moduleVersion;
216            moduleVersion = forGradleInfo(attributes);
217            if (moduleVersion != null)
218                return moduleVersion;
219
220            // However, some OTHER people also write incompatible data into the manifest.
221            moduleVersion = forMavenPluexus(attributes);
222            if (moduleVersion != null)
223                return moduleVersion;
224
225            moduleVersion = forMavenBundle(attributes);
226            if (moduleVersion != null)
227                return moduleVersion;
228
229            return null;
230        }
231    }

233    public static class ModuleMetadata {
234
235        private final ProductMetadata productMetadata;
236        private final ModuleVersion moduleVersion;
237        private final URL manifestUrl;
238        private final Manifest manifest;
239        private final Properties properties = new Properties();
240
241        public ModuleMetadata(
242                @Nonnull ProductMetadata productMetadata,
243                @Nonnull ModuleVersion moduleVersion,
244                @CheckForNull URL manifestUrl,
245                @Nonnull Manifest manifest) {
246            this.productMetadata = productMetadata;
247            this.moduleVersion = moduleVersion;
248            this.manifestUrl = manifestUrl;
249            this.manifest = manifest;
250        }
251
252        @Nonnull
253        public ProductMetadata getProductMetadata() {
254            return productMetadata;
255        }
256
257        @Nonnull
258        public ModuleVersion getModuleVersion() {
259            return moduleVersion;
260        }
261
262        /**
263         * Returns the name of this module.
264         *
265         * There is no particularly standard format for names; it might be
266         * a Maven coordinate name, or an OSGI name, or might even contain
267         * spaces. Caveat caller.
268         *
269         * @return the name of this module.
270         */
271        @Nonnull
272        public String getName() {
273            return getModuleVersion().toName();
274        }
275
276        /**
277         * Returns the version of this module.
278         *
279         * @return the version of this module.
280         */
281        @Nonnull
282        public String getVersion() {
283            return getModuleVersion().getVersion();
284        }
285
286        @Nonnull
287        public Manifest getManifest() {
288            return manifest;
289        }
290
291        @Nonnull
292        public Properties getProperties() {
293            return properties;
294        }
295
296        @CheckForNull
297        public File getJar() {
298            if (manifestUrl == null)
299                return null;
300            String resourcePath = manifestUrl.toString();
301            if (!resourcePath.startsWith("jar:file:"))
302                return null;
303            resourcePath = resourcePath.substring(9);
304            int idx = resourcePath.indexOf('!');
305            if (idx > 0)
306                resourcePath = resourcePath.substring(0, idx);
307            return new File(resourcePath.replace('/', File.separatorChar));
308        }
309
310        @CheckForNull
311        private String getMetadataProperty(@Nonnull String key) {
312            String value = properties.getProperty(key);
313            if (value == null)
314                value = manifest.getMainAttributes().getValue(key);
315            return value;
316        }
317
318        @CheckForNull
319        private String getMetadataProperty(@Nonnull MetadataProperty key) {
320            return getMetadataProperty(key.key);
321        }
322
323        /** @return The VCS revision of this module, as in Maven metadata. */
324        @CheckForNull
325        public String getRevision() {
326            return getMetadataProperty(MetadataProperty.Change);
327        }
328
329        @CheckForNull
330        public String getBuildBranch() {
331            return getMetadataProperty(MetadataProperty.Branch);
332        }
333
334        @CheckForNull
335        public String getBuildDate() {
336            // Build-Date=2015-01-01_10:09:09
337            // String text = getMetadataProperty("Build-Date");
338            // SimpleDateFormat format = new SimpleDateFormat("yyyy-MM-dd_HH:mm:ss");
339            return getMetadataProperty(MetadataProperty.Build_Date);
340        }
341
342        @Nonnull
343        public String getSummary() {
344            StringBuilder buf = new StringBuilder();
345            buf.append(getModuleVersion());
346            String branch = getBuildBranch();
347            if (branch != null && !"master".equals(branch))
348                buf.append(" (branch ").append(branch).append(")");
349            String revision = getRevision();
350            if (revision != null)
351                buf.append(" (rev ").append(revision).append(")");
352            String buildDate = getBuildDate();
353            if (buildDate != null)
354                buf.append(" built on ").append(buildDate);
355            return buf.toString();
356        }
357
358        @Override
359        public String toString() {
360            return getSummary();
361        }
362    }

364    @Nonnull
365    private static ClassLoader getDefaultClassLoader() {
366        ClassLoader loader = Thread.currentThread().getContextClassLoader();
367        if (loader != null)
368            return loader;
369        return ProductMetadata.class.getClassLoader();
370    }

372    @CheckForNull
373    private static Manifest newManifest(@Nonnull URL resource) {
374        try {
375            InputStream in = resource.openStream();
376            try {
377                return new Manifest(in);
378            } finally {
379                in.close();
380            }
381        } catch (Exception e) {
382            // This should never happen because we previously discovered the manifest as a resource.
383            // However, if we replace a JAR underneath a running JVM, we can get FileNotFoundException.
384            getLogger().log(Level.WARNING, "Failed to load Manifest from " + resource, e);
385            return null;
386        }
387    }

389    private final ClassLoader loader;
390    private final Map<String, ModuleMetadata> modules = new TreeMap<String, ModuleMetadata>();
391
392    /**
393     * Creates a new ProductMetadata.
394     *
395     * @param loader The ClassLoader which can access the product resources.
396     * @throws IOException Thrown only if the "global" load fails, making this object useless. The failure of any module will be ignored.
397     */
398    public ProductMetadata(@Nonnull ClassLoader loader) throws IOException {
399        this.loader = loader;
400        load();
401    }
402
403    /**
404     * Creates a new ProductMetadata.
405     *
406     * @throws IOException Thrown only if the "global" load fails, making this object useless. The failure of any module will be ignored.
407     */
408    public ProductMetadata() throws IOException {
409        this(getDefaultClassLoader());
410    }

412    @CheckForNull
413    private ModuleMetadata newModule(@Nonnull URL manifestUrl, @Nonnull Manifest manifest, @Nonnull ModuleVersion version) {
414        // URL resource = loader.getResource("META-INF/" + version.getName() + ".properties");
415        ModuleMetadata module = new ModuleMetadata(this, version, manifestUrl, manifest);
416        String spec = "/META-INF/" + version.getName() + ".properties";
417        try {
418            URL propertiesUrl = new URL(manifestUrl, spec);
419            // getLogger().info("Loading " + propertiesUrl);
420            try {
421                InputStream in = propertiesUrl.openStream();
422                try {
423                    module.properties.load(in);
424                } finally {
425                    in.close();
426                }
427                // LOG.debug("Loaded " + module + " from " + resource);   // Might throw.
428            } catch (FileNotFoundException e) {
429                // getLogger().log(Level.WARNING, "Failed to load ModuleMetadata from " + spec, e);
430            }
431        } catch (Exception e) {
432            getLogger().log(Level.WARNING, "Failed to load ModuleMetadata for " + version + " from " + spec, e);
433        }
434        return module;
435    }
436
437    private void load() throws IOException {
438        // This is a cheap way to discover JAR files.
439        Enumeration<URL> manifestUrls = loader.getResources("META-INF/MANIFEST.MF");
440        for (URL manifestUrl : Collections.list(manifestUrls)) {
441            // System.err.println("Found " + resource);
442            Manifest manifest = newManifest(manifestUrl);
443            if (manifest == null)
444                continue;
445            Attributes attributes = manifest.getMainAttributes();
446            if (attributes == null)
447                continue;
448            ModuleVersion moduleVersion = ModuleVersion.forAttributes(attributes);
449            if (moduleVersion == null)
450                continue;
451            ModuleMetadata moduleMetadata = newModule(manifestUrl, manifest, moduleVersion);
452            if (moduleMetadata == null)
453                continue;
454            modules.put(moduleMetadata.getName(), moduleMetadata);
455        }
456    }

458    /**
459     * Returns the ModuleMetadata for all discovered modules.
460     *
461     * @return the ModuleMetadata for all discovered modules.
462     */
463    @Nonnull
464    public Collection<ModuleMetadata> getModules() {
465        return new ArrayList<ModuleMetadata>(modules.values());
466    }
467
468    /**
469     * Retrieves the metadata for a module by name.
470     *
471     * Names are not particularly consistent - the best names are
472     * Maven coordinates derived from packages built using the
473     * gradle-info-plugin, but tolerable names
474     * are derived from OSGI packageers like Maven Plexus or Bnd plugins.
475     *
476     * @param name The module name, e.g. "org.anarres.jdiagnostics:jdiagnostics" or "slf4j-api".
477     * @return The ModuleMetadata, if found, else null.
478     */
479    @CheckForNull
480    public ModuleMetadata getModule(@Nonnull String name) {
481        return modules.get(name);
482    }

484    /**
485     * A utility for producing a support text listing only modules
486     * within the product.
487     *
488     * For example, toString("org.anarres.") will include all modules
489     * containing metadata with a name starting with "org.anarres." but
490     * exclude all others.
491     *
492     * @param namePrefix The prefix to apply to {@link ModuleMetadata#getName()}.
493     * @return An equivalent of {@link #toString()} with only requested modules.
494     */
495    @Nonnull
496    public String toString(String namePrefix) {
497        StringBuilder buf = new StringBuilder();
498        for (ModuleMetadata module : getModules())
499            if (module.getName().startsWith(namePrefix))
500                buf.append(module).append("\n");
501        return buf.toString();
502    }

504    @Override
505    public String toString() {
506        return toString("");
507    }
