001package org.anarres.jdiagnostics;
002
003import java.util.LinkedHashMap;
004
005/**
006 *
007 * @author shevek
008 */
009public class ClassLoaderQuery extends AbstractQuery {
010
011    private final String name;
012    private final ClassLoader loader;
013
014    public ClassLoaderQuery(String name, ClassLoader loader) {
015        this.name = name;
016        this.loader = loader;
017    }
018
019    @Override
020    public String getName() {
021        return "classLoader/" + name;
022    }
023
024    @Override
025    protected Result newResult() {
026        return new Result(new LinkedHashMap<String, Object>());
027    }
028
029    @Override
030    public void call(Result result, String prefix) {
031        ClassLoader l = loader;
032        int i = 0;
033        while (l != null) {
034            result.put("loaderClass[" + i + "]", l.getClass());
035            result.put("loader[" + i + "]", l);
036            l = l.getParent();
037            i++;
038        }
039    }
040
041}
